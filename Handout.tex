\section{Zeit der Bürgerkriege}
\begin{itemize}
	\item Betonung Roms im Gegensatz zu Marcus Antonius (z.B. Augustusmausoleum)
	\item Caesarkult (z.B. Curia Iulia, Tempel des Divus Iulius)
	\item Bezug zu Apollo
	\aitem Repräsentative Bauten in Rom
\end{itemize}

\section{Zeit des Prinzipats}
\begin{itemize}
	\item Fertigstellung und Sanierung vieler Bauwerke und der Infrastruktur (Brandschutz, Wasserversorgung, \dots)
	\item Bau und Erneuerung vieler Tempel
	\aitem Dabei Verwendung von teurem Marmor zur Repräsentation des weltweiten Einfluss
	\aitem Betonung der \glqq{}aetas aurea\grqq{}
	\aitem Dadurch ausgelöster \glqq{}Wettbewerb\grqq{} unter der Elite $\rightarrow$ Goldenes Zeitalter wird sichtbar
	\aitem Ziel: Darstellung als Friedensbringer (Schließung des Tempel des Ianus Quirinum, Bau der Ara Pacis, \dots)
\end{itemize}

\section{Ara Pacis Augustae}

\begin{itemize}
	\item Errichtet vom Senat wegen Augustus Rückkehr von Feldzügen in Spanien und Gallien
	\item Errichtet auf dem Marsfeld neben Augustusmausoleum und Ara Pacis
	\item Geweiht am Geburtstag seiner Frau Livia
	\item Aufbau ähnlich einem templum minus $\rightarrow$ Aber aus massivem lunensichem Marmor (= Carrara-Marmor)
	\item Heute als Rekonstruktion im Museo dell'Ara Pacis
\end{itemize}
